// 1번 문제: 이 학생의 점수는 84점이며, 등급은 B등급 입니다!
  // 조건
     1. 90점 ~ 100점: A등급
     2. 80점 ~ 89점 : B등급
     3. 80점 이하는 C등급
  // 출력 형태 : 이 학생의 점수는 [점수]이며, 등급은 [등급]입니다!

void main() {
  // 학생과 점수
  Map<String, int> Students = {
    'Anna': 30, 
    'Bob': 84, 
    'Casey': 96
  }; // 학생 별 각 부여된 점수를 키와 값으로 줌

  // 등급을 정하는 함수
  // 조건문 if / if else 의 구조를 사용해 비교연산자를 활용해 값의 범위 조건을 정하고 구현함 
  String grade (int score) {
    if (score >= 90) {
      return 'A';
    } else if (score >= 80) {
      return 'B';
    } else {
      return 'C';
    }
  } 

  // 학생의 점수와 등급을 출력
  // 문제풀이과정 문제점 : '학생'과 '점수'를 찾아와서 출력을 시켜야 하는데 이 함수를 어떻게 써야 할지 한참 헤맸다.
  Students.
  print ('이 $Students 학생의 점수는 $score점 이며, 등급은 $grade등급 입니다!');
}
